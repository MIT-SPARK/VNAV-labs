  roscpp
  rospy
  opencv3_catkin
  glog_catkin
  image_transport
  cv_bridge_catkin
)

 INCLUDE_DIRS include
 LIBRARIES feature_trackers
 CATKIN_DEPENDS roscpp rospy opencv3_catkin glog_catkin image_transport cv_bridge_catkin
)

include_directories(
  include
  ${catkin_INCLUDE_DIRS}
)
